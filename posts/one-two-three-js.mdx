import {
  Scene,
  PerspectiveCamera,
  WebGLRenderer,
  TorusKnotGeometry,
  MeshBasicMaterial,
  Mesh,
} from "three"

<!-- Set up scene, camera and renderer -->

export const scene = new Scene()
export const camera = new PerspectiveCamera(
  75,
  window.innerWidth / window.innherHeight,
  0.1,
  1000
)
export const renderer = WebGLRenderer()
renderer.setPixelRatio(window.devicePixelRatio)
renderer.setSize(400, 300)
document.body.appendChild(renderer.domElement)

<!-- Set up torus knot -->

export const geometry = TorusKnotGeometry(10, 3, 100, 16)
export const material = MeshBasicMaterial({ color: 0xffff00 })
export const torusKnot = Mesh(geometry, material)
scene.add(torusKnot)
camera.position.z = 5

<!-- Render the scene -->

export const animate = () => {
  requestAnimationFrame(animate)
  torusKnot.rotation.x += 0.01
  torusKnot.rotation.y += 0.01
  renderer.render(scene, camera)
}
animate()
